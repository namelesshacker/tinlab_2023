	void enqueue(id_t element)
	{
		list[len++] = element;
	}
	
	// Remove the front element of the queue
	void dequeue()
	{
		int i = 0;
		len -= 1;
		while (i < len)
		{
			list[i] = list[i + 1];
			i++;
		}
		list[i] = 0;
	}
	
	// Returns the front element of the queue
	id_t front()
	{
		return list[0];
	}
	
	// Returns the last element of the queue
	id_t tail()
	{
		return list[len - 1];
	}
